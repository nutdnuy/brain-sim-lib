**Goal:** Add Riskfolio-Lib-style tutorial examples for `brain-sim`, starting with an executable offline notebook that teaches Excel batch alpha simulation without requiring live BRAIN credentials.

**Tech Stack:** Python 3.10+, standard-library JSON/CSV/pathlib, `openpyxl`, existing `brain_sim` package APIs, pytest.

## Reference Pattern

The target style follows Riskfolio-Lib's examples surface:

- `examples/` is a first-class folder in the GitHub repo.
- Tutorial files are named with a numbered prefix such as `Tutorial 1 - Classic Mean Risk Optimization.ipynb`.
- Tutorial 1 uses a narrative notebook structure: title, setup/data section, then numbered workflow sections.
- The repository root README should point readers to the tutorial folder.

For `brain-sim`, the tutorial must not run live simulations by default. It should teach the workflow with a fake client, then show live commands as explicitly marked commands the user can run after login.

```python
from __future__ import annotations

import csv
import json
from pathlib import Path
from typing import Any

from openpyxl import Workbook


ROOT = Path(__file__).resolve().parents[1]
EXAMPLES = ROOT / "examples"
DATA_DIR = EXAMPLES / "data"
EXPECTED_DIR = EXAMPLES / "expected"
NOTEBOOK_PATH = EXAMPLES / "Tutorial 1 - Excel Batch Alpha Simulation.ipynb"
WORKBOOK_PATH = DATA_DIR / "tutorial_01_alphas.xlsx"
SUMMARY_PATH = EXPECTED_DIR / "tutorial_01_summary.csv"


ALPHA_ROWS = [
    ("tutorial-001", "rank(ts_mean(volume, 10))"),
    ("tutorial-002", "rank(ts_mean(volume, 20))"),
    ("tutorial-003", "rank(ts_mean(returns, 5))"),
    ("tutorial-004", "rank(ts_std_dev(returns, 20))"),
]


SUMMARY_FIELDS = [
    "row_id",
    "alpha_hash",
    "status",
    "alpha_id",
    "simulation_location",
    "sharpe",
    "fitness",
    "returns",
    "turnover",
    "drawdown",
    "margin",
    "longCount",
    "shortCount",
    "failed_checks",
    "warning_checks",
    "pending_checks",
    "error",
]


def ensure_dirs() -> None:
    DATA_DIR.mkdir(parents=True, exist_ok=True)
    EXPECTED_DIR.mkdir(parents=True, exist_ok=True)


def build_workbook() -> None:
    workbook = Workbook()
    sheet = workbook.active
    sheet.title = "alphas"
    sheet.append(
        [
            "id",
            "expression",
            "region",
            "universe",
            "delay",
            "decay",
            "neutralization",
            "truncation",
            "nanHandling",
            "language",
            "visualization",
        ]
    )
    for row_id, expression in ALPHA_ROWS:
        sheet.append(
            [
                row_id,
                expression,
                "USA",
                "TOP3000",
                1,
                5,
                "SUBINDUSTRY",
                0.08,
                "OFF",
                "FASTEXPR",
                False,
            ]
        )
    workbook.save(WORKBOOK_PATH)


def build_expected_summary() -> None:
    rows = []
    for index, (row_id, _expression) in enumerate(ALPHA_ROWS, start=1):
        rows.append(
            {
                "row_id": row_id,
                "alpha_hash": f"tutorial-hash-{index}",
                "status": "complete",
                "alpha_id": f"tutorial-alpha-{index}",
                "simulation_location": f"/simulations/tutorial-{index}",
                "sharpe": f"{1.00 + index / 10:.2f}",
                "fitness": f"{0.50 + index / 10:.2f}",
                "returns": f"{0.03 + index / 100:.2f}",
                "turnover": f"{0.10 + index / 100:.2f}",
                "drawdown": f"{0.02 + index / 100:.2f}",
                "margin": "",
                "longCount": "",
                "shortCount": "",
                "failed_checks": "",
                "warning_checks": "",
                "pending_checks": "",
                "error": "",
            }
        )
    with SUMMARY_PATH.open("w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=SUMMARY_FIELDS)
        writer.writeheader()
        writer.writerows(rows)


def markdown_cell(source: str) -> dict[str, Any]:
    return {
        "cell_type": "markdown",
        "metadata": {},
        "source": source.splitlines(keepends=True),
    }


def code_cell(source: str) -> dict[str, Any]:
    return {
        "cell_type": "code",
        "execution_count": None,
        "metadata": {},
        "outputs": [],
        "source": source.splitlines(keepends=True),
    }


def build_notebook() -> None:
    cells = [
        markdown_cell(
            "# Tutorial 1 - Excel Batch Alpha Simulation\n\n"
            "This tutorial mirrors the numbered notebook style used by mature quant libraries. "
            "It teaches the full `brain-sim` workflow with an offline fake BRAIN client, so readers can run it safely before using live credentials.\n"
        ),
        markdown_cell(
            "## 1. What This Tutorial Builds\n\n"
            "You will create a small Excel alpha queue, convert it into `brain-sim` payload records, run an offline batch simulation, and inspect the same artifacts produced by live runs: raw logs, `summary.csv`, and `retry_queue.jsonl`.\n"
        ),
        code_cell(
            "from __future__ import annotations\n\n"
            "import csv\n"
            "import shutil\n"
            "from dataclasses import dataclass\n"
            "from pathlib import Path\n"
            "from typing import Any\n\n"
            "from brain_sim.batch import BatchRunner\n"
            "from brain_sim.client import PollResult, SubmitResult\n"
            "from brain_sim.excel import read_excel_expressions\n"
            "from brain_sim.models import RateLimitState\n"
            "from brain_sim.payloads import build_payload_record\n\n"
            "DRY_RUN = True\n"
            "ROOT = Path.cwd().resolve()\n"
            "EXAMPLE_DIR = ROOT / 'examples'\n"
            "EXCEL_PATH = EXAMPLE_DIR / 'data' / 'tutorial_01_alphas.xlsx'\n"
            "RUN_DIR = ROOT / 'examples' / 'runs' / 'tutorial_01_offline'\n"
        ),
        markdown_cell(
            "## 2. Create A Sample Excel Queue\n\n"
            "The workbook uses the same shape expected by `brain-sim simulate-excel`: one required `expression` column plus optional settings columns. "
            "The sample file is already committed at `examples/data/tutorial_01_alphas.xlsx`.\n"
        ),
        code_cell(
            "expressions = read_excel_expressions(EXCEL_PATH, sheet_name='alphas')\n"
            "records = [build_payload_record(expression) for expression in expressions]\n"
            "[(record.row_id, record.payload['regular'], record.payload['settings']['universe']) for record in records]\n"
        ),
        markdown_cell(
            "## 3. Understand Batch Simulation Artifacts\n\n"
            "`BatchRunner` needs a client with `submit`, `poll`, `fetch_alpha`, and `fetch_recordset` methods. "
            "The fake client below returns deterministic alpha IDs and metrics while preserving the same control flow as a live run.\n"
        ),
        code_cell(
            "@dataclass\n"
            "class FakeTutorialBrainClient:\n"
            "    next_id: int = 0\n\n"
            "    def submit(self, payload: dict[str, Any] | list[dict[str, Any]]) -> SubmitResult:\n"
            "        self.next_id += 1\n"
            "        location = f'/simulations/tutorial-{self.next_id}'\n"
            "        return SubmitResult(\n"
            "            status_code=201,\n"
            "            location=location,\n"
            "            body={'id': location.rsplit('/', 1)[-1]},\n"
            "            headers={'Location': location},\n"
            "            rate_limit=RateLimitState(limit=5000, remaining=4999, reset_seconds=3600),\n"
            "        )\n\n"
            "    def poll(self, location: str, *, timeout_seconds: float) -> PollResult:\n"
            "        alpha_index = int(location.rsplit('-', 1)[-1])\n"
            "        alpha_id = f'tutorial-alpha-{alpha_index}'\n"
            "        return PollResult(\n"
            "            status='complete',\n"
            "            body={'status': 'COMPLETE', 'alpha': alpha_id},\n"
            "            events=[{'status_code': 200, 'body': {'status': 'COMPLETE', 'alpha': alpha_id}}],\n"
            "        )\n\n"
            "    def fetch_alpha(self, alpha_id: str) -> dict[str, Any]:\n"
            "        alpha_index = int(alpha_id.rsplit('-', 1)[-1])\n"
            "        return {\n"
            "            'id': alpha_id,\n"
            "            'status': 'UNSUBMITTED',\n"
            "            'is': {\n"
            "                'sharpe': round(1.00 + alpha_index / 10, 2),\n"
            "                'fitness': round(0.50 + alpha_index / 10, 2),\n"
            "                'returns': round(0.03 + alpha_index / 100, 2),\n"
            "                'turnover': round(0.10 + alpha_index / 100, 2),\n"
            "                'drawdown': round(0.02 + alpha_index / 100, 2),\n"
            "                'checks': [{'name': 'LOW_SHARPE', 'result': 'PASS'}],\n"
            "            },\n"
            "        }\n\n"
            "    def fetch_recordset(self, alpha_id: str, recordset_name: str) -> dict[str, Any]:\n"
            "        return {'alpha_id': alpha_id, 'recordset': recordset_name, 'rows': []}\n"
        ),
        markdown_cell(
            "## 4. Run The Offline Simulation\n\n"
            "This cell deletes only the tutorial offline run folder, then runs four single-item submissions. "
            "It does not authenticate or call the live BRAIN API.\n"
        ),
        code_cell(
            "if not DRY_RUN:\n"
            "    raise RuntimeError('This tutorial must run with DRY_RUN = True.')\n\n"
            "if RUN_DIR.exists():\n"
            "    shutil.rmtree(RUN_DIR)\n\n"
            "runner = BatchRunner(FakeTutorialBrainClient(), RUN_DIR)\n"
            "result = runner.run(records, batch_size=1, poll_timeout_seconds=5)\n"
            "result\n"
        ),
        markdown_cell(
            "## 5. Inspect Summary And Retry Queue\n\n"
            "Successful runs write `summary.csv`. Failed, pending, or partially collected runs also write `retry_queue.jsonl` with the original payload and simulation location when available.\n"
        ),
        code_cell(
            "summary_path = RUN_DIR / 'summary.csv'\n"
            "with summary_path.open(newline='', encoding='utf-8') as f:\n"
            "    summary_rows = list(csv.DictReader(f))\n\n"
            "[(row['row_id'], row['status'], row['alpha_id'], row['simulation_location']) for row in summary_rows]\n"
        ),
        markdown_cell(
            "## 6. Live BRAIN Run After Login\n\n"
            "Run these commands in a terminal only after you are ready to use live BRAIN quota:\n\n"
            "```bash\n"
            "brain-sim login --print-link --credentials-file ~/.brain_credentials\n"
            "brain-sim simulate-excel examples/data/tutorial_01_alphas.xlsx --sheet alphas --batch-size 4 --poll-timeout-seconds 1800\n"
            "```\n"
        ),
        markdown_cell(
            "## 7. Batch Timeout And Retry Rules\n\n"
            "If BRAIN accepts a batch but polling stays pending, `brain-sim` records `pending_timeout`. "
            "The retry queue keeps `simulation_location`, which lets a later tool inspect the accepted job before resubmitting the same alpha payload.\n"
        ),
    ]
    notebook = {
        "cells": cells,
        "metadata": {
            "kernelspec": {
                "display_name": "Python 3",
                "language": "python",
                "name": "python3",
            },
            "language_info": {
                "name": "python",
                "pygments_lexer": "ipython3",
            },
        },
        "nbformat": 4,
        "nbformat_minor": 5,
    }
    NOTEBOOK_PATH.write_text(json.dumps(notebook, indent=2, ensure_ascii=False) + "\n", encoding="utf-8")


def main() -> None:
    ensure_dirs()
    build_workbook()
    build_expected_summary()
    build_notebook()


if __name__ == "__main__":
    main()
